/**
 * on a interval function is ran to check for when the request is complete, then marks it complete,
 * can we conditionally do this?
 * if save an object with 0 processes and status === 'active'?
 * @param  {Object} input   parsed rabbitMQ pageScan message
 * @param  {Object} ch      rabbitMQ Channel Object
 * @param  {Object} timeOut setInterval object
 */
function pollRequest(input, ch, timeOut) {
     let deferred = q.defer();
     utils.findBy(Request, {
          requestId: input.requestId
     }, function (err, data) {
          if (data && data.status === 'running' && (data.processes === 0 || data.processes < 0)) {
               utils.completeRequest(deferred, input);
               deferred.promise.catch(retry);
               ch.ack(true);
               clearInterval(timeOut);
          } else {
               deferred.resolve(true);
          }
     });
}
