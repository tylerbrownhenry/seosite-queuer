function checkRequirements(requirements, input) {
     var promise = q.defer();
     var passed = true;
     var _params = [];
     _.each(requirements, function (key) {
          if (typeof input[key] === 'undefined' || input[key] === '') {
               passed = false;
               _params.push(key);
          }
     });
     if (passed === true) {
          promise.resolve();
     } else {
          var messages = [];
          _.each(_params, function (param) {
               messages.push({
                    parent: param,
                    title: 'Oops!',
                    param:param,
                    message: 'error:missing:required:param'
               })
          });
          promise.reject({
               success: false,
               type: 'userInput',
               message: messages
          });
     }
     return promise.promise;
}
