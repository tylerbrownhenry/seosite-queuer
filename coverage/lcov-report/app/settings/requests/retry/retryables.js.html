function batchPut(promise, opts) {
     let batchPut = require('../../../utils').batchPut,
          model = require("../../../models/metaData");
     if (opts.model === 'Link') {
          model = require("../../../models/link");
     } else if (opts.model === 'Resource') {
          model = require("../../../models/resource");
     }
     batchPut(model, opts.commands, (err) => {
          if (err) {
               promise.reject(err);
          } else {
               retry({
                    retryCommand: 'utils.updateBy',
                    retryOptions: {
                         model: 'Request',
                         input: {
                              id: opts.requestId
                         },
                         update: {
                              processes: -1
                         }
                    }
               });
               promise.resolve();
          }
     });
     return promise.promise;
}
 
function updateBy(promise, opts) {
     let dynamoose = require('dynamoose'),
          updateBy = require('../../../utils').updateBy;
     if (opts.model === 'Request') {
          var requestSchema = require("../../../models/request"),
               Request = dynamoose.model('Request', requestSchema);
          updateBy(Request, opts.input, opts.update, (err) => {
               if (err) {
                    promise.reject(err);
               } else {
                    promise.resolve();
               }
          });
     } else if (opts.model === 'W3Cvalidation') {
          var W3Cvalidation = require("../../../models/W3Cvalidation");
          updateBy(W3Cvalidation, opts.input, opts.update, (err) => {
               if (err) {
                    promise.reject(err);
               } else {
                    promise.resolve();
               }
          });
     } else if (opts.model === 'Scan') {
          var Scan = require("../../../models/scan");
          updateBy(Scan, opts.input, opts.update, (err) => {
               if (err) {
                    promise.reject(err);
               } else {
                    promise.resolve();
                    if (opts.updateRequestProcess === true) {
                         let deferred = q.defer();
                         utils.retryUpdateRequest({
                              requestId: opts.requestId
                         }, deferred);
                         deferred.promise.catch(retry);
                    }
               }
          });
     } else {
          console.log('retrayables.js updateBy --> Not a supported model', opts.model)
          promise.reject();
     }
     return promise.promise;
}
