let utils = require('../../../../utils'),
     Scan = require('../../../../models/scan'),
     actions = require("../../actions").publish,
     q = require("q"),
     _ = require("underscore");
/**
 * after scanning the page, parse the html for information
 * @param  {Object} input request info and options
 * @param  {Object} res   information returned from page scan
 */
function processHar(input, res) {
     let url = input.url,
          options = input.options,
          deferred = input.promise,
          requestId = input.requestId,
          uid = input.uid,
          source = input.source,
          processes = 1, /* w3c was published */
          isRetry = input.isRetry;
 
     res.emails = input.emails; /* TODO Double check if this is a length and if Scan has it */
     res.completedTime = utils.getNowUTC();
 
     res.uid = uid;
     res.tapTargetCheck = res.checkElements;
     res.requestId = requestId;
 
     let newScan = new Scan(res);
 
     const arrActions = [ /*'screenCapture',*/ 'checkLinks', 'checkSSL', 'checkSocial', 'checkServerInfo', 'checkResource'];
     var processActions = [processMetaData(newScan, input, res)];
     _.each(arrActions, function (key) {
          processActions.push(actions({
               action: key,
               input: input,
               res: res,
               newScan: newScan
          }));
     });
 
     q.all(processActions).then((resp) => {
          console.log('processActions', resp);
          _.each(resp, function (re) {
               processes += re.processes
          });
 
          utils.saveScan(newScan, (err) => {
               if (err !== null) {
                    return deferred.reject({
                         processes:processes,
                         message: 'error:save:scan',
                         retry: true,
                         retryCommand: 'request.pageScan.processHar',
                         retryOptions: {
                              res: res,
                              input: input
                         }
                    });
               } else {
                    deferred.resolve({
                         processes:processes,
                         notify: true
                    });
               }
          })
     }).catch((e) => {
          return deferred.reject(e);
     })
     return deferred.promise;
}
 
module.exports = processHar;
