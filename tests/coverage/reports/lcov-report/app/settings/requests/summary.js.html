     var promise = q.defer();
     var input = preFlight(promise, msg, reject);
     if (input === false) {
          return promise.promise;

          console.log('request/summary.js init -> checkRequirements : failed');
          reject(promise, 'input', 'Missing required fields', true, input, page, false)
